//2. To read a string and convert alternative character into upper case.

#include<stdio.h>
#include<string.h>
int main()
{
    char str[50];
    int i;
    printf("Enter string: ");
    gets(str);
    for(i=0;str[i]!='\0';i++)
    {
        if((i%2)==0)
            str[i]=str[i]-32;
    }
    printf("%s",str);
    return 0;
}

INPUT:
Enter string: sujitha
OUTPUT:
SuJiThA
